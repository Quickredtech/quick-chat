from flask import Flask, render_template, request, redirect, url_for, session
from flask_socketio import SocketIO, emit, join_room, leave_room
from flask import render_template_string
import os

app = Flask(__name__)
app.config['SECRET_KEY'] = 'your_secret_key_here'
socketio = SocketIO(app)

# Simple in-memory storage for users and messages
users = set()
messages = []

CHAT_TEMPLATE = '''
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Flask Web Chat</title>
    <style>
        body { font-family: Arial, sans-serif; background: #f4f4f4; }
        #chat { width: 400px; margin: 40px auto; background: #fff; padding: 20px; border-radius: 8px; box-shadow: 0 2px 8px #ccc; }
        #messages { height: 300px; overflow-y: scroll; border: 1px solid #ddd; padding: 10px; margin-bottom: 10px; background: #fafafa; }
        #user { margin-bottom: 10px; }
        .msg { margin-bottom: 8px; }
        .user { font-weight: bold; color: #0074d9; }
        .me { color: #2ecc40; }
    </style>
</head>
<body>
<div id="chat">
    <h2>Flask Web Chat</h2>
    <div id="user">
        <input type="text" id="username" placeholder="Enter your name" autocomplete="off" />
        <button onclick="setUsername()">Join</button>
    </div>
    <div id="messages"></div>
    <form id="msgform" onsubmit="return sendMessage();" style="display:none;">
        <input id="message" autocomplete="off" placeholder="Type a message..." />
        <button>Send</button>
    </form>
</div>
<script src="https://cdnjs.cloudflare.com/ajax/libs/socket.io/4.7.5/socket.io.min.js"></script>
<script>
    var socket = io();
    var username = '';
    function setUsername() {
        username = document.getElementById('username').value.trim();
        if(username) {
            document.getElementById('user').style.display = 'none';
            document.getElementById('msgform').style.display = '';
            socket.emit('join', username);
        }
    }
    function sendMessage() {
        var msg = document.getElementById('message').value;
        if(msg.trim()) {
            socket.emit('message', {user: username, msg: msg});
            document.getElementById('message').value = '';
        }
        return false;
    }
    socket.on('message', function(data) {
        var messages = document.getElementById('messages');
        var div = document.createElement('div');
        div.className = 'msg';
        div.innerHTML = '<span class="user">' + data.user + ':</span> ' + data.msg;
        messages.appendChild(div);
        messages.scrollTop = messages.scrollHeight;
    });
    socket.on('history', function(msgs) {
        var messages = document.getElementById('messages');
        messages.innerHTML = '';
        msgs.forEach(function(data) {
            var div = document.createElement('div');
            div.className = 'msg';
            div.innerHTML = '<span class="user">' + data.user + ':</span> ' + data.msg;
            messages.appendChild(div);
        });
        messages.scrollTop = messages.scrollHeight;
    });
</script>
</body>
</html>
'''

@app.route('/')
def index():
    return render_template_string(CHAT_TEMPLATE)

@socketio.on('join')
def on_join(username):
    users.add(username)
    emit('history', messages)

@socketio.on('message')
def handle_message(data):
    messages.append({'user': data['user'], 'msg': data['msg']})
    emit('message', data, broadcast=True)

if __name__ == '__main__':
    socketio.run(app, debug=True)
